`백준-2193번 이친수`  
`핵심 ↓`  
![my alternate text](/assets/boj_2193.PNG)  
dp[i] = dp[N-1] + dp[N-2];  
주의! N의 범위를 생각해 변수 크기를 고려할 것.  

`소스코드 ↓`  
{% highlight cpp %}
#include <iostream>
#include <vector>

using namespace std;

int N;
vector<unsigned long long> dp;

void Func()
{
	dp[1] = 1;
	for (int i = 2; i <= N; i++)
	{
		dp[i] = dp[i - 1] + dp[i - 2];
	}
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cin >> N;
	dp = vector<unsigned long long>(N + 1, 0);
	Func();
	cout << dp[N] << "\n";
	return 0;
}
